#include <Wire.h>
#include <Adafruit_GFX.h>
#include <Adafruit_SSD1306.h>
#include <Servo.h>

// --- OLED setup ---
#define SCREEN_WIDTH 128
#define SCREEN_HEIGHT 64
#define OLED_RESET -1
Adafruit_SSD1306 display(SCREEN_WIDTH, SCREEN_HEIGHT, &Wire, OLED_RESET);

// --- Pin definitions ---
#define LED_PIN 8
#define BUTTON_RED 1
#define BUTTON_BLACK 2
#define BUTTON_YELLOW 3
#define BUTTON_BLUE 4
#define TRIG_PIN 9
#define ECHO_PIN 10
#define SERVO1_PIN 6
#define SERVO2_PIN 7

// --- Objects ---
Servo servo1;
Servo servo2;

// --- Variables ---
long duration;
int distance;
String buttonPressed = "None";

void setup() {
  Serial.begin(9600);

  // LED setup
  pinMode(LED_PIN, OUTPUT);

  // Buttons setup
  pinMode(BUTTON_RED, INPUT_PULLUP);
  pinMode(BUTTON_BLACK, INPUT_PULLUP);
  pinMode(BUTTON_YELLOW, INPUT_PULLUP);
  pinMode(BUTTON_BLUE, INPUT_PULLUP);

  // Ultrasonic sensor setup
  pinMode(TRIG_PIN, OUTPUT);
  pinMode(ECHO_PIN, INPUT);

  // Servo setup
  servo1.attach(SERVO1_PIN);
  servo2.attach(SERVO2_PIN);
  servo1.write(90); // start centered
  servo2.write(90);

  // OLED setup
  if(!display.begin(SSD1306_SWITCHCAPVCC, 0x3C)) {
    Serial.println("SSD1306 allocation failed");
    for(;;);
  }
  display.clearDisplay();
  display.setTextSize(2);
  display.setTextColor(SSD1306_WHITE);
  display.setCursor(0,0);
  display.println("Ready!");
  display.display();
  delay(1000);
}

void loop() {
  // --- Read ultrasonic distance ---
  digitalWrite(TRIG_PIN, LOW);
  delayMicroseconds(2);
  digitalWrite(TRIG_PIN, HIGH);
  delayMicroseconds(10);
  digitalWrite(TRIG_PIN, LOW);

  duration = pulseIn(ECHO_PIN, HIGH);
  distance = duration * 0.034 / 2; // convert to cm

  // --- Control servos and LED ---
  if(distance > 0 && distance < 15) { // hand detected
    servo1.write(0);   // move left
    servo2.write(180); // move right
    digitalWrite(LED_PIN, HIGH);
  } else { // hand gone
    servo1.write(90);  // center
    servo2.write(90);  // center
    digitalWrite(LED_PIN, LOW);
  }

  // --- Read buttons ---
  if(digitalRead(BUTTON_RED) == LOW) buttonPressed = "Aluminum";
  else if(digitalRead(BUTTON_BLACK) == LOW) buttonPressed = "Plastic";
  else if(digitalRead(BUTTON_YELLOW) == LOW) buttonPressed = "General";
  else if(digitalRead(BUTTON_BLUE) == LOW) buttonPressed = "Glass";
  else buttonPressed = "None";

  // --- Display on OLED ---
  display.clearDisplay();
  display.setCursor(0,0);
  display.setTextSize(2);
  display.println(buttonPressed);
  display.display();

  delay(100); // small delay for stability
}
